6. Reusable Code Snippets

6.1 Vocabulary class

import re
from collections import Counter

class Vocabulary:
    PAD, START, END, UNK = "<pad>", "<start>", "<end>", "<unk>"

    def __init__(self):
        self.itos, self.stoi = {}, {}
        for tok in [self.PAD, self.START, self.END, self.UNK]:
            self._add(tok)

    def _add(self, tok):
        idx = len(self.itos)
        self.itos[idx] = tok
        self.stoi[tok] = idx

    @staticmethod
    def tokenize(text):
        return re.findall(r"[a-z']+", text.lower())

    def build(self, captions, min_freq=5):
        counter = Counter()
        for c in captions:
            counter.update(self.tokenize(c))
        for word, freq in counter.items():
            if freq >= min_freq:
                self._add(word)

    def encode(self, text):
        return [self.stoi.get(t, self.stoi[self.UNK]) for t in self.tokenize(text)]

    def __len__(self):
        return len(self.itos)

6.2 Model (full)

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision.models as models

class EncoderCNN(nn.Module):
    def __init__(self):
        super().__init__()
        resnet = models.resnet50(weights=models.ResNet50_Weights.DEFAULT)
        modules = list(resnet.children())[:-2]
        self.cnn = nn.Sequential(*modules)
        for p in self.cnn.parameters():
            p.requires_grad = False

    def forward(self, images):
        f = self.cnn(images)                          # (B, 2048, 7, 7)
        f = f.permute(0, 2, 3, 1).flatten(1, 2)       # (B, 49, 2048)
        return f

class BahdanauAttention(nn.Module):
    def __init__(self, enc_dim=2048, dec_dim=512, attn_dim=512):
        super().__init__()
        self.W_a = nn.Linear(dec_dim, attn_dim, bias=False)
        self.U_a = nn.Linear(enc_dim, attn_dim, bias=False)
        self.v_a = nn.Linear(attn_dim, 1, bias=False)

    def forward(self, features, hidden):
        scores = self.v_a(torch.tanh(
            self.U_a(features) + self.W_a(hidden).unsqueeze(1)
        )).squeeze(-1)                                # (B, 49)
        alpha = F.softmax(scores, dim=1)
        context = (alpha.unsqueeze(-1) * features).sum(dim=1)
        return context, alpha

class DecoderWithAttention(nn.Module):
    def __init__(self, vocab_size, emb_dim=256, dec_dim=512, enc_dim=2048):
        super().__init__()
        self.attention = BahdanauAttention(enc_dim, dec_dim)
        self.embedding = nn.Embedding(vocab_size, emb_dim)
        self.lstm = nn.LSTMCell(emb_dim + enc_dim, dec_dim)
        self.init_h = nn.Linear(enc_dim, dec_dim)
        self.init_c = nn.Linear(enc_dim, dec_dim)
        self.fc = nn.Linear(dec_dim, vocab_size)
        self.dropout = nn.Dropout(0.5)

    def init_hidden(self, features):
        m = features.mean(dim=1)
        return torch.tanh(self.init_h(m)), torch.tanh(self.init_c(m))

    def forward(self, features, captions):
        B, T = captions.shape
        h, c = self.init_hidden(features)
        embeds = self.embedding(captions)
        logits, alphas = [], []
        for t in range(T):
            ctx, alpha = self.attention(features, h)
            inp = torch.cat([embeds[:, t], ctx], dim=1)
            h, c = self.lstm(inp, (h, c))
            logits.append(self.fc(self.dropout(h)))
            alphas.append(alpha)
        return torch.stack(logits, 1), torch.stack(alphas, 1)

6.3 Training step (the loss)

criterion = nn.CrossEntropyLoss(ignore_index=vocab.stoi[Vocabulary.PAD])
LAMBDA_DS = 1.0   # doubly-stochastic weight

for images, caps in loader:
    feats = encoder(images.to(device))
    logits, alphas = decoder(feats, caps[:, :-1].to(device))   # input: [start, w1, ..., wT-1]
    targets = caps[:, 1:].to(device)                           # target: [w1, ..., wT-1, end]
    loss_xe = criterion(logits.reshape(-1, logits.size(-1)), targets.reshape(-1))
    loss_ds = ((1.0 - alphas.sum(dim=1)) ** 2).mean()
    loss = loss_xe + LAMBDA_DS * loss_ds

    optimizer.zero_grad()
    loss.backward()
    nn.utils.clip_grad_norm_(decoder.parameters(), 5.0)
    optimizer.step()

6.4 Beam search (compact version)

@torch.no_grad()
def beam_search(encoder, decoder, image, vocab, beam=5, max_len=20):
    feats = encoder(image)                       # (1, 49, 2048)
    h, c = decoder.init_hidden(feats)
    start = vocab.stoi[Vocabulary.START]
    end = vocab.stoi[Vocabulary.END]

    beams = [(0.0, [start], h, c, [])]           # (logprob, tokens, h, c, alphas)
    finished = []

    for _ in range(max_len):
        new_beams = []
        for lp, toks, h, c, alphas in beams:
            if toks[-1] == end:
                finished.append((lp, toks, alphas))
                continue
            emb = decoder.embedding(torch.tensor([[toks[-1]]], device=image.device)).squeeze(1)
            ctx, alpha = decoder.attention(feats, h)
            h2, c2 = decoder.lstm(torch.cat([emb, ctx], dim=1), (h, c))
            log_probs = F.log_softmax(decoder.fc(h2), dim=-1).squeeze(0)
            top_lp, top_idx = log_probs.topk(beam)
            for k in range(beam):
                new_beams.append((
                    lp + top_lp[k].item(),
                    toks + [top_idx[k].item()],
                    h2, c2,
                    alphas + [alpha.squeeze(0).cpu().numpy()],
                ))
        beams = sorted(new_beams, key=lambda x: x[0], reverse=True)[:beam]
        if not beams:
            break

    finished += [(lp, t, a) for lp, t, _, _, a in beams]
    finished.sort(key=lambda x: x[0] / max(len(x[1]), 1), reverse=True)
    return finished[0]   # (logprob, token_ids, list of alpha arrays)

6.5 Grad-CAM

class GradCAM:
    def __init__(self, model, target_layer):
        self.model = model
        self.activations, self.gradients = None, None
        target_layer.register_forward_hook(
            lambda m, i, o: setattr(self, "activations", o.detach()))
        target_layer.register_full_backward_hook(
            lambda m, gi, go: setattr(self, "gradients", go[0].detach()))

    def __call__(self, image, target_logit):
        self.model.zero_grad()
        target_logit.backward(retain_graph=True)
        alpha = self.gradients.mean(dim=(2, 3), keepdim=True)
        cam = F.relu((alpha * self.activations).sum(dim=1, keepdim=True))
        cam = F.interpolate(cam, size=image.shape[-2:], mode="bilinear", align_corners=False)
        cam = cam - cam.min()
        cam = cam / (cam.max() + 1e-8)
        return cam.squeeze().cpu().numpy()

10. Risk Plan вҖ” What If Things Go Wrong?

RiskMitigation
Training takes too long, no GPU available Use Google Colab (T4) or Kaggle (P100) free tier; both have 12-hour sessions.
Loss diverges / captions are gibberish Reduce LR to 1e-4, lower λ on doubly-stochastic, verify <pad> masking is correct.
Trained model is too heavy for Streamlit Cloud (free tier ~1 GB RAM) Use ResNet-34 instead of ResNet-50, or quantize to fp16; or host on HF Spaces (more memory).
BLEU low (<0.10) Train longer, try teacher-forcing ratio scheduling, double-check vocabulary build is on TRAIN split only.
Grad-CAM heatmap is all-zero Make sure target_logit was computed with requires_grad=True in the path; never wrap inference in torch.no_grad for Grad-CAM.
No time for deployment Local-only demo is acceptable if you record a clean GIF; deployment can come post-deadline.
